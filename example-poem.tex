    \subsubsection{Medián}\label{mediuxe1n}

    \textbf{Medián} je prostřední hodnota v souboru. Pokud je počet hodnot v
souboru sudý je medián průměrem 2 prostředních hodnot.

\[ med(x) = 4 \]

    \subsubsection{Modus}\label{modus}

    \textbf{Modus} je hodnota s největší četností

\[ mod(x) = 2 \]

    \subsubsection{Průměr}\label{prux16fmux11br}

    \[ {\bar{x}} = \frac{\sum_{i=1}^{n}}{n} \]
\[ {\bar{x}} = \frac{1*5 + 2*28 + 3*19..}{113} = \frac{471}{113} = 4,168141593 \]

    \subsubsection{Rozptyl}\label{rozptyl}

    \[ Var(x) = \frac{5*(1−\bar{x})^2 + 28*(2−\bar{x})^2+19*(3 − \bar{x})^2 +21*(4−\bar{x})^2+...}{113} = 5,320  \]

    \subsubsection{Směrodatná odchylka}\label{smux11brodatnuxe1-odchylka}

    \[ \sigma = \sqrt{Var(x)} = \sqrt{5,320} = 2,206 \]

    \subsubsection{Variační koeficient}\label{variaux10dnuxed-koeficient}

    \[ v = \frac{\sigma}{\bar{x}} * 100\% = \frac{2,206}{4,168} * 100\% = 55,335 \% \]
